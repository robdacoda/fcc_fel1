  function getRandomInt(max) {

    return Math.floor(Math.random() * max);

  }


  function getQuote() {

    let x = getRandomInt(30);

    let y = getRandomInt(200);

    if (x < y) {

      return bigLongString.slice(x, y);

    } else {

      return bigLongString.slice(y, x);

    }

  }

  console.log(getQuote());

  let authorName = [];

  authorName = ["Fred", "Sue", "Joe", "Random Man", "Molly"];

  function getAuthorName() {

    return authorName[Math.round(getRandomInt(5))];

  }


  function getStuff() {

    document.getElementById("text").innerText = getQuote();

    document.getElementById("author").innerText = "-  " + getAuthorName();

  }
